15void IRAM_ATTR Anemometer::onPulseIsr() {
16 uint32_t now = millis();
17 if (now - s_lastPulseMs < kDebounceMs) {
18 return; // reed switch bounce, ignore

20 s_lastPulseMs = now;

30bool Anemometer::read(JsonObject &out) {
31 uint32_t total = 0;
32 uint32_t maxBin = 0;
33
34 // Count pulses in 1-second bins: the average gives "ws", the busiest
35 // bin gives the gust "wg".
36 for (uint8_t i = 0; i < _windowS; i++) {
37 s_binPulses = 0;
38 delay(1000);
39 uint32_t bin = s_binPulses;
40 total += bin;
41 maxBin = max(maxBin, bin);
42 }
43
44 double avgHz = (double)total / (double)_windowS;
45 double ws = avgHz * (double)_mpsPerHz;
46 double wg = (double)maxBin * (double)_mpsPerHz;
47
48 log_d("wind: %lu pulses in %u s -> avg %.1f m/s (%.1f km/h), gust %.1f m/s",
49 (unsigned long)total, _windowS, ws, ws * 3.6, wg);
50
51 out["ws"] = round(ws * 10.0) / 10.0;
52 out["wg"] = round(wg * 10.0) / 10.0;
53 return true;
54}

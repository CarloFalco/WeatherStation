25bool AppConfig::load() {
26 File file = LittleFS.open(kConfigPath, "r");
27 if (!file) {
28 return false;
29 }
30
31 String section;
32 while (file.available()) {
33 String line = file.readStringUntil('\n');
34 line.trim();
35
36 // Skip blanks and full-line comments.
37 if (line.isEmpty() || line[0] == ';' || line[0] == '#') {
38 continue;
39 }
40
41 // Section header: [name]
42 if (line[0] == '[' && line.endsWith("]")) {
43 section = line.substring(1, line.length() - 1);
44 section.trim();
45 section.toLowerCase();
46 continue;
47 }
48
49 int eq = line.indexOf('=');
50 if (eq < 0) {
51 log_w("config.ini: ignoring malformed line '%s'", line.c_str());
52 continue;
53 }
54
55 String key = line.substring(0, eq);
56 String value = line.substring(eq + 1);
57 key.trim();
58 key.toLowerCase();
59 value.trim();
60
61 applyKey(section, key, value);
62 }
63 file.close();
64 return true;
65}

67void AppConfig::applyKey(const String &section, const String &key, const String &value) {
68 if (section == "station") {
69 if (key == "id") {
70 station.id = value;
71 } else if (key == "wake_interval_s") {
72 long v = value.toInt();
73 if (v >= 10) {
74 station.wakeIntervalS = (uint32_t)v;
75 } else {
76 log_w("config.ini: wake_interval_s=%ld too small (<10 s), keeping %lu",
77 v, (unsigned long)station.wakeIntervalS);
78 }
79 } else {
80 log_w("config.ini: unknown key [station] %s", key.c_str());
81 }
82 return;
83 }
84
85 if (section == "rain") {
86 if (key == "mm_per_pulse") {
87 float v = value.toFloat();
88 if (v > 0.0f && v < 10.0f) {
89 rain.mmPerPulse = v;
90 } else {
91 log_w("config.ini: mm_per_pulse=%s out of range (0..10), keeping %.4f",
92 value.c_str(), rain.mmPerPulse);
93 }
94 } else {
95 log_w("config.ini: unknown key [rain] %s", key.c_str());
96 }
97 return;
98 }
99
100 if (section == "wind") {
101 if (key == "mps_per_hz") {
102 float v = value.toFloat();
103 if (v > 0.0f && v < 10.0f) {
104 wind.mpsPerHz = v;
105 } else {
106 log_w("config.ini: mps_per_hz=%s out of range (0..10), keeping %.3f",
107 value.c_str(), wind.mpsPerHz);
108 }
109 } else if (key == "sample_window_s") {
110 long v = value.toInt();
111 wind.sampleWindowS = (uint8_t)constrain(v, 1L, 30L);
112 } else if (key == "vane_offset_deg") {
113 long v = value.toInt();
114 // Normalize any input to 0..359.
115 wind.vaneOffsetDeg = (int16_t)(((v % 360) + 360) % 360);
116 } else {
117 log_w("config.ini: unknown key [wind] %s", key.c_str());
118 }
119 return;
120 }
121
122 if (section == "power") {
123 if (key == "shunt_mohm") {
124 long v = value.toInt();
125 if (v >= 1 && v <= 1000) {
126 power.shuntMohm = (uint32_t)v;
127 } else {
128 log_w("config.ini: shunt_mohm=%ld out of range (1..1000), keeping %lu",
129 v, (unsigned long)power.shuntMohm);
130 }
131 } else {
132 log_w("config.ini: unknown key [power] %s", key.c_str());
133 }
134 return;
135 }
136
137 if (section == "ota") {
138 if (key == "chunk_timeout_ms") {
139 long v = value.toInt();
140 ota.chunkTimeoutMs = (uint16_t)constrain(v, 200L, 10000L);
141 } else if (key == "max_retries") {
142 long v = value.toInt();
143 ota.maxRetries = (uint8_t)constrain(v, 1L, 50L);
144 } else if (key == "session_timeout_s") {
145 long v = value.toInt();
146 ota.sessionTimeoutS = (uint16_t)constrain(v, 30L, 3600L);
147 } else {
148 log_w("config.ini: unknown key [ota] %s", key.c_str());
149 }
150 return;
151 }
152
153 if (section == "lora") {
154 if (key == "freq_mhz") {
155 lora.freqMhz = value.toFloat();
156 } else if (key == "bw_khz") {
157 lora.bwKhz = value.toFloat();
158 } else if (key == "sf") {
159 long v = value.toInt();
160 lora.sf = (uint8_t)constrain(v, 6L, 12L);
161 } else if (key == "cr") {
162 long v = value.toInt();
163 lora.cr = (uint8_t)constrain(v, 5L, 8L);
164 } else if (key == "tx_power_dbm") {
165 long v = value.toInt();
166 lora.txPowerDbm = (int8_t)constrain(v, 2L, 17L);
167 } else if (key == "sync_word") {
168 // Accepts both decimal and 0x-prefixed hex.
169 lora.syncWord = (uint8_t)strtol(value.c_str(), nullptr, 0);
170 } else if (key == "ack_enabled") {
171 lora.ackEnabled = (value == "true" || value == "1" || value == "yes");
172 } else if (key == "ack_timeout_ms") {
173 long v = value.toInt();
174 lora.ackTimeoutMs = (uint16_t)constrain(v, 100L, 5000L);
175 } else if (key == "tx_retries") {
176 long v = value.toInt();
177 lora.txRetries = (uint8_t)constrain(v, 0L, 5L);
178 } else {
179 log_w("config.ini: unknown key [lora] %s", key.c_str());
180 }
181 return;
182 }
183
184 log_w("config.ini: unknown section [%s]", section.c_str());
185}

187bool AppConfig::save() const {
188 File file = LittleFS.open(kConfigPath, "w");
189 if (!file) {
190 log_e("Cannot open %s for writing", kConfigPath);
191 return false;
192 }
193
194 file.println("; WeatherStation V2 - runtime configuration");

197 file.println();
198 file.println("[station]");
199 file.printf("id = %s\n", station.id.c_str());

209 file.println();
210 file.println("[power]");
211 file.printf("shunt_mohm = %lu\n", (unsigned long)power.shuntMohm);
212 file.println();
213 file.println("[lora]");
214 file.printf("freq_mhz = %.1f\n", lora.freqMhz);
215 file.printf("bw_khz = %.1f\n", lora.bwKhz);
216 file.printf("sf = %u\n", lora.sf);
217 file.printf("cr = %u\n", lora.cr);

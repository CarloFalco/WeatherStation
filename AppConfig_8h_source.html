1/**
2 * @file AppConfig.h
3 * @brief Runtime configuration stored as an INI file on LittleFS.
4 *
5 * Hardware constants (pins, I2C addresses) live in include/config.h and are
6 * fixed at build time; everything the user may want to tune in the field
7 * (station identity, wake interval, LoRa radio parameters) lives here and
8 * is loaded from /config.ini at every boot.
9 *
10 * If the file is missing (first boot, erased flash) the defaults defined
11 * in this class are used and a fresh /config.ini is written back, so the
12 * filesystem image upload (`pio run -t uploadfs`) is optional.
13 *
14 * Usage example:
15 * @code
16 * AppConfig config;
17 * config.begin(); // mount FS + load (or create) file
18 * uint64_t us = config.station.wakeIntervalS * 1000000ULL;
19 * @endcode
20 */
21

25#include <Arduino.h>

27/**
28 * @brief Loads, stores and persists the station runtime configuration.
29 *
30 * Minimal hand-rolled INI parser (sections, key=value, ';'/'#' comments):
31 * no external library needed for a file this small.
32 */
33class AppConfig {
34public:
35 /** @brief `[station]` section: identity and duty cycle. */
36 struct StationConfig {
37 String id = "ws-01"; ///< Station identifier sent in every LoRa message.
38 uint32_t wakeIntervalS = 600; ///< Seconds of deep sleep between measurement cycles.
39 };
40
41 /** @brief `[rain]` section: rain-gauge calibration. */
42 struct RainConfig {
43 float mmPerPulse = 0.4743f; ///< Rainfall per bucket tip [mm]: bucket
44 ///< volume / collector area = 3 cm3 / 63.25 cm2.
45 };
46
47 /** @brief `[wind]` section: anemometer calibration and vane mounting offset. */
48 struct WindConfig {
49 float mpsPerHz = 0.667f; ///< Wind speed per pulse frequency [m/s per Hz]
50 ///< (cup anemometer convention "1 Hz = 2.4 km/h").
51 uint8_t sampleWindowS = 5; ///< Wind sampling window per cycle [s].
52 int16_t vaneOffsetDeg = 0; ///< Added to the AS5600 raw angle to align 0 to North.
53 };
54
55 /** @brief `[power]` section: INA3221 energy monitor. */
56 struct PowerConfig {
57 uint32_t shuntMohm = 100; ///< Shunt resistance [milliohm], all channels
58 ///< (hardware-verified against a reference
59 ///< current during Increment 6 validation).
60 };
61
62 /** @brief `[ota]` section: firmware transfer tuning (docs/lora-protocol.md). */
63 struct OtaConfig {
64 uint16_t chunkTimeoutMs = 1500; ///< Wait for a chunk after each request [ms].
65 uint8_t maxRetries = 8; ///< Attempts per chunk before giving up.
66 uint16_t sessionTimeoutS = 1800; ///< Cap on the awake time of one session [s].
67 ///< A full image is ~2400 chunks / ~20 min:
68 ///< an interrupted session resumes later.
69 };
70
71 /** @brief `[lora]` section: SX1276 radio parameters (see docs/lora-protocol.md). */
72 struct LoraConfig {
73 float freqMhz = 868.1f; ///< Carrier frequency [MHz], EU868 band.
74 float bwKhz = 125.0f; ///< Bandwidth [kHz].
75 uint8_t sf = 7; ///< Spreading factor (7..12).
76 uint8_t cr = 5; ///< Coding rate denominator (5 -> 4/5).
77 int8_t txPowerDbm = 14; ///< TX power [dBm], EU868 ERP limit.
78 uint8_t syncWord = 0x12; ///< Private-network sync word.
79 bool ackEnabled = true; ///< Wait for the base ACK after each TX.
80 uint16_t ackTimeoutMs = 600; ///< RX window length after the TX [ms].
81 uint8_t txRetries = 1; ///< Retransmissions when the ACK is missing.
82 };
83
84 /**
85 * @brief Mount LittleFS and load /config.ini.
86 *
87 * On a missing file the defaults are kept and persisted with save().
88 *
89 * @return true if the filesystem is available (config is usable),
90 * false on unrecoverable filesystem failure (defaults still set).
91 */
92 bool begin();
93
94 /**
95 * @brief Parse /config.ini and override the current values.
96 * @return true if the file existed and was parsed.
97 */
98 bool load();
99
100 /**
101 * @brief Write the current configuration back to /config.ini.
102 * @return true on success.
103 */
104 bool save() const;
105
106 /**
107 * @brief Pretty-print the active configuration (for the serial log).
108 * @param out Destination stream, e.g. Serial.
109 */
110 void printTo(Stream &out) const;
111
112 StationConfig station; ///< Active `[station]` values.
113 RainConfig rain; ///< Active `[rain]` values.
114 WindConfig wind; ///< Active `[wind]` values.
115 PowerConfig power; ///< Active `[power]` values.
116 LoraConfig lora; ///< Active `[lora]` values.
117 OtaConfig ota; ///< Active `[ota]` values.
118
119private:
120 /// Apply a single parsed `key = value` pair belonging to @p section.
121 void applyKey(const String &section, const String &key, const String &value);
122
123 static constexpr const char *kConfigPath = "/config.ini"; ///< File location on LittleFS.
124};

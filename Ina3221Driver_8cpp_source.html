11bool Ina3221Driver::readReg(uint8_t reg, uint16_t &val) {
12 Wire.beginTransmission(_addr);
13 Wire.write(reg);
14 if (Wire.endTransmission(false) != 0) { // repeated start
15 return false;
16 }
17 if (Wire.requestFrom(_addr, (uint8_t)2) != 2) {
18 return false;
19 }
20 val = ((uint16_t)Wire.read() << 8) | Wire.read();
21 return true;
22}

48bool Ina3221Driver::waitConversion(uint32_t timeoutMs) {
49 uint32_t start = millis();
50 uint16_t mask = 0;
51 while (millis() - start < timeoutMs) {
52 if (readReg(kRegMaskEnable, mask) && (mask & kMaskConvReady)) {
53 return true;
54 }
55 delay(5);
56 }
57 log_w("INA3221: conversion-ready timeout");
58 return false;

61float Ina3221Driver::busVoltageMv(uint8_t channel) {
62 uint16_t raw = 0;
63 if (channel >= 3 || !readReg(kRegBusV[channel], raw)) {
64 return NAN;
65 }
66 // 13-bit value in bits 15..3, LSB = 8 mV: (raw >> 3) * 8 == raw with
67 // the low bits masked off.
68 return (float)((int16_t)raw / 8 * 8);
69}

71float Ina3221Driver::currentMa(uint8_t channel) {
72 uint16_t raw = 0;
73 if (channel >= 3 || !readReg(kRegShuntV[channel], raw)) {
74 return NAN;
75 }
76 // Signed 13-bit value in bits 15..3, LSB = 40 uV. Dividing the int16_t
77 // by 8 (not shifting the unsigned raw) preserves the sign.
78 int32_t shuntUv = (int32_t)((int16_t)raw / 8) * 40;
79 // I [mA] = V [uV] / R [mOhm] (1e-6 / 1e-3 = 1e-3, i.e. mA directly)
80 return (float)shuntUv / (float)_shuntMohm;

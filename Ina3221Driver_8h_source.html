1/**
2 * @file Ina3221Driver.h
3 * @brief Minimal register-level driver for the TI INA3221 current monitor.
4 *
5 * Derived from the Beast Devices INA3221 Arduino library
6 * (MIT License, Copyright (c) 2020 Beast Devices, Andrejs Bondarevs),
7 * stripped down and adapted for this project:
8 * - single-shot (triggered) conversions instead of continuous mode: the
9 * chip powers itself down after each conversion set (~0.35 mA saved
10 * between measurement cycles, which matters on a >12-month battery
11 * budget);
12 * - SIGNED shunt voltage reading: the INA3221 shunt ADC is bidirectional
13 * (13-bit two's complement) so charge/discharge current comes out with
14 * its real sign — the original code read it unsigned and needed
15 * workarounds downstream;
16 * - everything not used by the station (alerts, limits, sum channels,
17 * filter-resistor offset estimation) removed.
18 */
19

26/**
27 * @brief Register-level access to the INA3221 (3-channel bus/shunt monitor).
28 */
29class Ina3221Driver {
30public:
31 /**
32 * @param i2cAddr 7-bit I2C address (0x40 with A0 to GND).
33 */
34 explicit Ina3221Driver(uint8_t i2cAddr) : _addr(i2cAddr) {}
35
36 /**
37 * @brief Probe the chip on the bus.
38 *
39 * The global Wire bus must already be initialized.
40 *
41 * @return true if the manufacturer ID register reads 0x5449 ("TI").
42 */
43 bool begin();
44
45 /**
46 * @brief Set the shunt resistor value used for current conversion.
47 * @param mohm Shunt resistance [milliohm], same on all three channels.
48 */
49 void setShuntResMohm(uint32_t mohm) { _shuntMohm = mohm; }
50
51 /**
52 * @brief Start one single-shot conversion of all three channels.
53 *
54 * Configuration: 16-sample averaging, 1.1 ms conversion time for both
55 * bus and shunt (~106 ms total). The chip returns to power-down by
56 * itself when the conversion set completes.
57 */
58 bool triggerConversion();
59
60 /**
61 * @brief Block until the conversion-ready flag is set.
62 * @param timeoutMs Give-up timeout [ms].
63 * @return true if the conversion completed within the timeout.
64 */
65 bool waitConversion(uint32_t timeoutMs = 500);
66
67 /**
68 * @brief Bus (load-side) voltage of a channel.
69 * @param channel Channel index 0..2.
70 * @return Voltage [mV] (8 mV LSB), or NAN on I2C error.
71 */
72 float busVoltageMv(uint8_t channel);
73
74 /**
75 * @brief Signed current through a channel shunt.
76 *
77 * Positive when conventional current flows from IN+ to IN-.
78 *
79 * @param channel Channel index 0..2.
80 * @return Current [mA], or NAN on I2C error.
81 */
82 float currentMa(uint8_t channel);
83
84private:
85 bool readReg(uint8_t reg, uint16_t &val);
86 bool writeReg(uint8_t reg, uint16_t val);
87
88 // Register map (subset).
89 static constexpr uint8_t kRegConf = 0x00;
90 static constexpr uint8_t kRegShuntV[3] = {0x01, 0x03, 0x05};
91 static constexpr uint8_t kRegBusV[3] = {0x02, 0x04, 0x06};
92 static constexpr uint8_t kRegMaskEnable = 0x0F;
93 static constexpr uint8_t kRegManufId = 0xFE;
94
95 static constexpr uint16_t kManufIdTi = 0x5449; ///< "TI" in ASCII.
96 /// All channels enabled, AVG=16, CT bus/shunt=1.1 ms, mode=single-shot.
97 static constexpr uint16_t kConfSingleShot = 0x7523;
98 static constexpr uint16_t kMaskConvReady = 0x0001; ///< CVRF flag.
99
100 uint8_t _addr; ///< 7-bit I2C address.
101 uint32_t _shuntMohm = 100; ///< Shunt resistance [milliohm].
102};

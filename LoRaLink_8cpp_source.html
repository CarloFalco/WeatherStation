55bool LoRaLink::receive(String &payload, uint32_t timeoutMs) {
56 if (!_ready) {
57 return false;
58 }
59
60 // RadioLib blocking receive() uses RX-single mode with a ~100-symbol
61 // timeout (~100 ms at SF7): poll it until our own deadline expires.
62 uint32_t deadline = millis() + timeoutMs;
63 while ((int32_t)(deadline - millis()) > 0) {
64 int16_t state = _radio.receive(payload);
65 if (state == RADIOLIB_ERR_NONE) {
66 log_d("LoRa RX: %u bytes, RSSI %.0f dBm, SNR %.1f dB",
67 (unsigned)payload.length(), _radio.getRSSI(), _radio.getSNR());
68 return true;
69 }
70 if (state != RADIOLIB_ERR_RX_TIMEOUT) {
71 log_w("LoRa RX error %d (corrupted packet?)", state);
72 // CRC mismatch or similar: keep listening until the deadline.
73 }
74 }
75 return false;
76}

78bool LoRaLink::receiveRaw(uint8_t *buf, size_t bufSize, size_t &len, uint32_t timeoutMs) {
79 if (!_ready || bufSize < RADIOLIB_SX127X_MAX_PACKET_LENGTH) {
80 return false;
81 }
82
83 // Continuous RX, polled on DIO0 (RxDone). The blocking receive() used
84 // for ACKs listens in RX-single windows of ~100 symbols (~102 ms at
85 // SF7) and stops between them: during an OTA transfer the base answers
86 // within tens of milliseconds, and a chunk whose preamble fell into one
87 // of those gaps was lost, costing a retry. In continuous mode the radio
88 // never stops listening until we take it out.
89 int16_t state = _radio.startReceive();
90 if (state != RADIOLIB_ERR_NONE) {
91 log_w("LoRa startReceive failed, error %d", state);
92 return false;
93 }
94
95 uint32_t deadline = millis() + timeoutMs;
96 while ((int32_t)(deadline - millis()) > 0) {
97 if (digitalRead(LORA_DIO0)) { // RxDone asserted
98 size_t pktLen = _radio.getPacketLength();
99 state = _radio.readData(buf, pktLen);
100 if (state == RADIOLIB_ERR_NONE) {
101 len = pktLen;
102 log_d("LoRa RX raw: %u bytes, RSSI %.0f dBm", (unsigned)len,
103 _radio.getRSSI());
104 _radio.standby();
105 return true;
106 }
107 log_w("LoRa RX raw error %d (corrupted packet?)", state);
108 _radio.startReceive(); // keep listening until the deadline
109 }
110 delay(1);
111 }
112
113 _radio.standby();
114 return false;
115}

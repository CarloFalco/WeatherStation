1/**
2 * @file LoRaLink.h
3 * @brief ITelemetryLink implementation for the SX1276 LoRa radio (RadioLib).
4 *
5 * Raw LoRa point-to-point (not LoRaWAN) on the EU 868 MHz band. All radio
6 * parameters come from config.ini ([lora] section) and must match the
7 * receiver configuration: see docs/lora-protocol.md.
8 *
9 * Transmission is blocking; after each telemetry TX main.cpp opens a
10 * short receive window waiting for the base ACK (see the "protocollo"
11 * section of docs/lora-protocol.md). Both behaviours are driven from
12 * config.ini ([lora] ack_enabled / ack_timeout_ms / tx_retries).
13 */

25 * @brief SX1276 telemetry link driven through RadioLib.
26 */
27class LoRaLink : public ITelemetryLink {
28public:
29 /**
30 * @param cfg Radio parameters (kept by reference: read at begin(),

32 */
33 explicit LoRaLink(const AppConfig::LoraConfig &cfg)
34 : _cfg(cfg), _module(LORA_CS, LORA_DIO0, LORA_RST, RADIOLIB_NC),
35 _radio(&_module) {}

37 /**
38 * @brief Initialize SPI and the SX1276 with the configured parameters.
39 * @return true if the radio answered and accepted the configuration.
40 */
41 bool begin() override;
42
43 /**
44 * @brief Transmit the payload (blocking, ~120 ms at SF7 for ~150 B).
45 * @param payload Compact JSON telemetry message.
46 * @return true on successful transmission.
47 */
48 bool send(const String &payload) override;
49
50 /**
51 * @brief Listen for one packet (RX single mode, polled until timeout).
52 * @param payload Filled with the received message on success.
53 * @param timeoutMs Maximum listening time [ms].
54 * @return true if a packet passed CRC and was read within the timeout.
55 */
56 bool receive(String &payload, uint32_t timeoutMs) override;
57
58 /**
59 * @brief Listen for one raw binary packet (OTA chunks).
60 *
61 * LoRa-specific (not part of ITelemetryLink): the OTA transfer uses
62 * raw frames because base64-in-JSON would waste ~33% of the scarce
63 * 255-byte LoRa payload.
64 *
65 * @param buf Destination buffer; MUST hold a full LoRa packet (>= 256 B).
66 * @param bufSize Size of @p buf.
67 * @param len Set to the received packet length on success.
68 * @param timeoutMs Maximum listening time [ms].
69 * @return true if a packet passed CRC and fit the buffer.
70 */
71 bool receiveRaw(uint8_t *buf, size_t bufSize, size_t &len, uint32_t timeoutMs);
72
73 /**
74 * @brief Put the SX1276 into sleep mode (~0.2 uA).
75 */
76 void sleep() override;
77
78private:
79 const AppConfig::LoraConfig &_cfg; ///< Radio parameters from config.ini.
80 Module _module; ///< RadioLib pin binding.
81 SX1276 _radio; ///< RadioLib radio driver.
82 bool _ready = false; ///< begin() outcome.
83};

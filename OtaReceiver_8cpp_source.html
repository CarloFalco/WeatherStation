41bool OtaReceiver::runSession(const Offer &offer) {
42 // Sanity checks on the offer before spending airtime.
43 if (offer.size == 0 || offer.chunks == 0 ||
44 (uint32_t)offer.chunks * kChunkData < offer.size ||
45 (uint32_t)(offer.chunks - 1) * kChunkData >= offer.size) {
46 log_e("OTA: inconsistent offer (size=%lu, chunks=%u)",
47 (unsigned long)offer.size, offer.chunks);
48 return false;
49 }
50
51 // Resume only if the base is offering exactly the image we already
52 // started, and the file on disk is as long as our progress claims.
53 bool resuming = g_rtcState.ota.size == offer.size &&
54 g_rtcState.ota.crc == offer.crc &&
55 g_rtcState.ota.chunks == offer.chunks &&
56 g_rtcState.ota.nextChunk > 0 &&
57 g_rtcState.ota.nextChunk < offer.chunks;
58
59 uint16_t startChunk = 0;
60 uint32_t crc = logic::crc32Init();
61
62 if (resuming) {
63 File probe = LittleFS.open(kImagePath, "r");
64 uint32_t onDisk = probe ? probe.size() : 0;
65 if (probe) {
66 probe.close();
67 }
68 if (onDisk == (uint32_t)g_rtcState.ota.nextChunk * kChunkData) {
69 startChunk = g_rtcState.ota.nextChunk;
70 crc = g_rtcState.ota.runningCrc;
71 } else {
72 log_w("OTA: resume point %u does not match file size %lu, restarting",
73 g_rtcState.ota.nextChunk, (unsigned long)onDisk);
74 resuming = false;
75 }
76 }
77
78 if (!resuming) {
79 g_rtcState.ota.size = offer.size;
80 g_rtcState.ota.crc = offer.crc;
81 g_rtcState.ota.chunks = offer.chunks;
82 saveProgress(0, logic::crc32Init());
83 }
84
85 File file = LittleFS.open(kImagePath, resuming ? "a" : "w");
86 if (!file) {
87 log_e("OTA: cannot open %s for writing", kImagePath);
88 return false;
89 }
90
91 if (resuming) {
92 Serial.printf("OTA transfer: RESUME from chunk %u/%u (%lu bytes, version %s)\n",
93 startChunk, offer.chunks, (unsigned long)offer.size,
94 offer.version.c_str());
95 } else {
96 Serial.printf("OTA transfer: %lu bytes in %u chunks (version %s)\n",
97 (unsigned long)offer.size, offer.chunks, offer.version.c_str());
98 }
99
100 uint32_t sessionStart = millis();
101 uint32_t sessionCapMs = (uint32_t)_sessionTimeoutS * 1000UL;
102 uint8_t buf[RADIOLIB_SX127X_MAX_PACKET_LENGTH + 1];
103 uint16_t idx = startChunk;
104 bool ok = true;
105
106 for (; idx < offer.chunks && ok; idx++) {
107 size_t expected = offer.size - (uint32_t)idx * kChunkData;
108 if (expected > kChunkData) {
109 expected = kChunkData;
110 }
111
112 for (uint8_t attempt = 1;; attempt++) {
113 if (millis() - sessionStart > sessionCapMs) {
114 log_e("OTA: session time cap (%u s) reached at chunk %u",
115 _sessionTimeoutS, idx);
116 ok = false;
117 break;
118 }
119 if (attempt > _maxRetries) {
120 log_e("OTA: chunk %u failed after %u attempts", idx, _maxRetries);
121 ok = false;
122 break;
123 }
124 if (attempt > 1) {
125 // Short randomized backoff: avoids locking into a
126 // request/chunk collision pattern with the base.
127 delay(20 + random(0, 60));
128 }
129
130 if (!sendChunkRequest(idx)) {
131 continue;
132 }
133 size_t len = 0;
134 if (!_link.receiveRaw(buf, sizeof(buf), len, _chunkTimeoutMs)) {
135 log_w("OTA: chunk %u timeout (attempt %u)", idx, attempt);
136 continue;
137 }
138
139 uint16_t rxIdx = (uint16_t)buf[1] | ((uint16_t)buf[2] << 8);
140 if (len != expected + kChunkHeader || buf[0] != kChunkMagic || rxIdx != idx) {
141 log_w("OTA: bad chunk (len=%u, idx=%u, expected %u/%u)",
142 (unsigned)len, rxIdx, (unsigned)(expected + kChunkHeader), idx);
143 continue;
144 }
145
146 file.write(buf + kChunkHeader, expected);
147 crc = logic::crc32Update(crc, buf + kChunkHeader, expected);
148 break;
149 }
150
151 if (ok && ((idx + 1) % 32 == 0 || idx + 1 == offer.chunks)) {
152 uint32_t elapsedS = (millis() - sessionStart) / 1000;
153 uint32_t done = idx + 1 - startChunk;
154 uint32_t etaS = done ? (uint32_t)((uint64_t)elapsedS *
155 (offer.chunks - idx - 1) / done)
156 : 0;
157 Serial.printf("OTA: %u/%u chunks (%u%%), elapsed %lu s, ETA %lu s\n",
158 idx + 1, offer.chunks,
159 (unsigned)(100UL * (idx + 1) / offer.chunks),
160 (unsigned long)elapsedS, (unsigned long)etaS);
161 }
162 }
163 file.flush();
164 file.close();
165
166 // idx is the first chunk NOT stored: that is exactly the resume point.
167 saveProgress(idx, crc);
168
169 uint32_t finalCrc = logic::crc32Final(crc);
170 bool success = ok && idx == offer.chunks && finalCrc == offer.crc;
171 if (ok && !success) {
172 log_e("OTA: CRC mismatch (got %08lX, expected %08lX)",
173 (unsigned long)finalCrc, (unsigned long)offer.crc);
174 }
175
176 sendReport(success, finalCrc);
177
178 if (success) {
179 g_rtcState.ota = {}; // transfer complete: nothing to resume
180 Serial.printf("OTA transfer SUCCESS: %lu bytes in %lu s (CRC %08lX)\n",
181 (unsigned long)offer.size,
182 (unsigned long)((millis() - sessionStart) / 1000),
183 (unsigned long)finalCrc);
184 } else {
185 Serial.printf("OTA transfer INCOMPLETE: %u/%u chunks stored, "
186 "will resume next cycle\n",
187 idx, offer.chunks);
188 }
189 return success;
190}

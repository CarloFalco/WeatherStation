1/**
2 * @file OtaReceiver.h
3 * @brief Chunked file transfer over LoRa — stage 1 of the OTA pipeline.
4 *
5 * Implements the receiver side of the OTA transfer protocol described in
6 * docs/lora-protocol.md. Pull model: the station requests every chunk
7 * explicitly (`ota_req`), so acknowledgement, ordering and retransmission
8 * are implicit — a lost or corrupted chunk is simply requested again.
9 *
10 * A full firmware image (~425 kB) is ~2400 chunks and takes ~20 minutes
11 * at SF7, so an interrupted session must not start over: progress
12 * (next chunk + running CRC) lives in RTC RAM and the transfer resumes
13 * on a later wake-up as soon as the base offers the same image again.
14 *
15 * This stage (fw 3.0.0-alpha.2) stores the received image on LittleFS and
16 * verifies its CRC-32. The next stages will stream into the inactive OTA
17 * app partition (esp_ota_*), reboot with rollback, and verify a firmware
18 * signature.
19 */

28/**

30 */
31class OtaReceiver {
32public:
33 /** @brief Transfer metadata announced by the base inside the ACK. */
34 struct Offer {
35 uint32_t size = 0; ///< Total image size [bytes].
36 uint32_t crc = 0; ///< Expected CRC-32 of the whole image.
37 uint16_t chunks = 0; ///< Number of chunks (ceil(size / 180)).
38 String version; ///< Version advertised by the base (informative).
39 };

41 /**
42 * @param link Radio link (LoRa-specific: raw chunks bypass JSON).
43 * @param stationId Station identifier put into every request.
44 */
45 OtaReceiver(LoRaLink &link, const String &stationId)
46 : _link(link), _id(stationId) {}
47
48 /**
49 * @brief Apply the runtime tuning from config.ini ([ota] section).
50 * @param chunkTimeoutMs Wait for a chunk after each request [ms].
51 * @param maxRetries Attempts per chunk before giving up.
52 * @param sessionTimeoutS Cap on the awake time of one session [s].
53 */
54 void configure(uint16_t chunkTimeoutMs, uint8_t maxRetries, uint16_t sessionTimeoutS) {
55 _chunkTimeoutMs = chunkTimeoutMs;
56 _maxRetries = maxRetries;
57 _sessionTimeoutS = sessionTimeoutS;
58 }
59
60 /**
61 * @brief Run a transfer session (blocking), resuming if possible.
62 *
63 * Requests every missing chunk, streams it to LittleFS, verifies the
64 * final CRC-32 and reports the outcome to the base with `ota_done`.
65 * On failure the progress is kept in RTC RAM so the next session
66 * continues from where this one stopped.
67 *
68 * @param offer Transfer metadata from the base ACK.
69 * @return true if the image was fully received and the CRC matches.
70 */
71 bool runSession(const Offer &offer);
72
73 /** @return Path of the received image on LittleFS. */
74 static const char *imagePath() { return kImagePath; }
75
76private:

79 static constexpr size_t kChunkData = 180; ///< Payload bytes per chunk.
80 static constexpr const char *kImagePath = "/ota_image.bin";
81
82 /** @brief Send the `ota_req` JSON for chunk @p idx. */
83 bool sendChunkRequest(uint16_t idx);
84
85 /** @brief Send the final `ota_done` report. */

87
88 /** @brief Persist the resume point in RTC RAM. */

97};

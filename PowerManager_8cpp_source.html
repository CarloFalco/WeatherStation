24const char *PowerManager::wakeupCauseString() const {
25 switch (_cause) {
26 case ESP_SLEEP_WAKEUP_TIMER: return "timer (periodic wake-up)";
27 case ESP_SLEEP_WAKEUP_EXT0: return "external signal (EXT0, rain gauge)";
28 case ESP_SLEEP_WAKEUP_EXT1: return "external signal (EXT1)";
29 case ESP_SLEEP_WAKEUP_ULP: return "ULP coprocessor";
30 case ESP_SLEEP_WAKEUP_UNDEFINED:
31 default: return "power-on / reset (cold boot)";
32 }
33}

35void PowerManager::deepSleep(uint32_t seconds) {
36 esp_sleep_enable_timer_wakeup((uint64_t)seconds * 1000000ULL);
37
38 // Rain-gauge wake-up: a bucket tip pulls RAIN_GAUGE_PIN low. The RTC
39 // domain needs its own pull-up because the digital GPIO one is powered
40 // down in deep sleep.
41 rtc_gpio_pullup_en(RAIN_GAUGE_PIN);
42 rtc_gpio_pulldown_dis(RAIN_GAUGE_PIN);
43 esp_sleep_enable_ext0_wakeup(RAIN_GAUGE_PIN, 0 /* wake on low */);
44
45 Serial.flush();
46 esp_deep_sleep_start();
47
48 // esp_deep_sleep_start() does not return; satisfy [[noreturn]] anyway.
49 while (true) {}
50}

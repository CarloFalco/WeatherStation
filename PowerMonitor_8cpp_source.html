14bool PowerMonitor::read(JsonObject &out) {
15 if (!_driver.triggerConversion() || !_driver.waitConversion()) {
16 return false;
17 }
18
19 float vbatMv = _driver.busVoltageMv(kChBattery);
20 float ibatMa = _driver.currentMa(kChBattery);
21 float ipanMa = _driver.currentMa(kChPanel);
22 float iloadMa = _driver.currentMa(kChLoad);
23
24 // Debug dump of the raw channel readings (serial monitor only, stripped
25 // at compile time when CORE_DEBUG_LEVEL < 4). Panel and load voltages
26 // are read inside the macro so they cost nothing in production builds.
27 log_d("panel : %6.3f V %8.1f mA", _driver.busVoltageMv(kChPanel) / 1000.0f, ipanMa);
28 log_d("battery: %6.3f V %8.1f mA (soc %d%%)", vbatMv / 1000.0f, ibatMa,
29 logic::vbatToSoc(vbatMv));
30 log_d("load : %6.3f V %8.1f mA", _driver.busVoltageMv(kChLoad) / 1000.0f, iloadMa);
31
32 if (isnan(vbatMv) || isnan(ibatMa) || isnan(ipanMa) || isnan(iloadMa)) {
33 return false;
34 }
35 if (vbatMv < kMinValidVbatMv) {
36 log_w("PowerMonitor: battery rail at %.0f mV, readings discarded", vbatMv);
37 return false;
38 }
39
40 // Sign convention (docs/lora-protocol.md): ibat positive = charging.
41 // With this board's wiring the shunt reads discharge as positive,
42 // hence the inversion.
43 out["vbat"] = round((double)vbatMv / 10.0) / 100.0; // V, 2 decimals
44 out["soc"] = logic::vbatToSoc(vbatMv);
45 out["ibat"] = (int)lround(-ibatMa);
46 out["ipan"] = (int)lround(ipanMa);
47 out["iload"] = (int)lround(iloadMa);
48 return true;
49}

1/**
2 * @file PowerMonitor.h
3 * @brief Energy monitoring via INA3221: panel, battery and load channels.
4 *
5 * Contributes the power-related telemetry fields:
6 * - "vbat" battery voltage [V]
7 * - "soc" battery state of charge [%], from the LiPo discharge curve
8 * (pure logic in src/logic/battery_soc.*, native-tested)
9 * - "ibat" battery current [mA], positive = charging
10 * - "ipan" solar panel current [mA]
11 * - "iload" load (ESP + sensors) current [mA]
12 *
13 * Channel mapping (fixed by the board wiring, verified on hardware during
14 * the Increment 6 validation — the old prototype comment had panel and
15 * load swapped):
16 * CH1 = load, CH2 = battery, CH3 = solar panel.
17 *
18 * Each cycle runs one single-shot conversion (the INA3221 sits in
19 * power-down the rest of the time). The shunt resistance is configurable
20 * in config.ini ([power] shunt_mohm).
21 */
22

11/// Timestamp of the last accepted pulse, for the debounce lockout.
12static volatile uint32_t s_lastPulseMs = 0;

14void IRAM_ATTR RainGauge::onPulseIsr() {
15 uint32_t now = millis();
16 if (now - s_lastPulseMs < kDebounceMs) {
17 return; // reed switch bounce, ignore
18 }
19 s_lastPulseMs = now;
20 g_rtcState.rainPulses++;
21}

29bool RainGauge::read(JsonObject &out) {

32 // Round to 2 decimals in double math to avoid float serialization

42void RainGauge::resetAccumulator() {
43 // Subtract the reported snapshot under a brief interrupt lock: the ISR
44 // may increment the counter concurrently and read-modify-write is not
45 // atomic. Tips landed after read() stay in the accumulator.
46 noInterrupts();
47 g_rtcState.rainPulses -= _reportedPulses;
48 interrupts();
49 _reportedPulses = 0;
50}

52void RainGauge::countSleepPulse() {

12bool WindVane::readRegisters(uint8_t reg, uint8_t *buf, size_t len) {
13 Wire.beginTransmission(AS5600_ADDRESS);
14 Wire.write(reg);
15 if (Wire.endTransmission(false) != 0) { // repeated start

17 }

19 return false;
20 }
21 for (size_t i = 0; i < len; i++) {
22 buf[i] = Wire.read();

27bool WindVane::begin() {
28 uint8_t status = 0;
29 if (!readRegisters(kRegStatus, &status, 1)) {
30 return false; // chip not answering on the bus
31 }
32 if (!(status & kStatusMagnetDetected)) {
33 log_w("AS5600: no magnet detected (status=0x%02X)", status);
34 return false;
35 }
36
37 // Switch to LPM3: the chip stays powered through deep sleep and would
38 // otherwise draw ~6.5 mA around the clock. The CONF register is
39 // volatile, so this runs on every wake-up (cheap: skipped when the
40 // bits are already set, e.g. woken from deep sleep, not power-on).
41 uint8_t conf[2];
42 if (readRegisters(kRegConf, conf, 2)) {
43 uint8_t lowByte = (conf[1] & ~kConfPowerModeMask) | kConfPowerModeLpm3;
44 if (lowByte != conf[1]) {
45 Wire.beginTransmission(AS5600_ADDRESS);
46 Wire.write(kRegConf);
47 Wire.write(conf[0]);
48 Wire.write(lowByte);
49 if (Wire.endTransmission() == 0) {
50 log_i("AS5600 switched to LPM3 (~1.5 mA instead of ~6.5 mA)");
51 } else {
52 log_w("AS5600: failed to set LPM3 power mode");
53 }
54 }
55 }
56 return true;
57}

59bool WindVane::read(JsonObject &out) {
60 uint8_t raw[2];
61 if (!readRegisters(kRegRawAngle, raw, 2)) {
62 return false;
63 }
64
65 uint16_t counts = ((uint16_t)(raw[0] & 0x0F) << 8) | raw[1]; // 12 bit
66 double rawDeg = (double)counts * (360.0 / 4096.0);
67
68 int wd = ((int)lround(rawDeg) + _offsetDeg) % 360;
69 if (wd < 0) {
70 wd += 360;
71 }
72
73 // 16-point compass rose, 22.5 deg per sector, N centered on 0.
74 // Only referenced by log_d: unused in production builds (level < 4).
75 [[maybe_unused]] static const char *kCompass[16] = {
76 "N", "NNE", "NE", "ENE", "E", "ESE", "SE", "SSE",
77 "S", "SSW", "SW", "WSW", "W", "WNW", "NW", "NNW"};
78 log_d("AS5600 raw %.1f deg -> wd %d deg (%s)", rawDeg, wd,
79 kCompass[((wd * 2 + 22) / 45) % 16]);
80
81 out["wd"] = wd;
82 return true;
83}

1/**
2 * @file WindVane.h
3 * @brief Wind direction vane based on the AS5600 magnetic encoder (I2C).
4 *
5 * Contributes the "wd" field: wind direction in degrees, 0-359, where
6 * 0 = North and angles grow clockwise (90 = East). The AS5600 reads the
7 * absolute angle of the magnet on the vane shaft; since the sensor can
8 * be mounted in any orientation, a configurable offset ([wind]
9 * vane_offset_deg in config.ini) maps the raw angle to true North:
10 *
11 * wd = (raw_angle_deg + vane_offset_deg) mod 360
12 *
13 * Field calibration: point the vane North and set vane_offset_deg to
14 * (360 - raw) using the raw angle logged on serial.
15 *
16 * The register protocol is trivial (two-byte reads), so the chip is
17 * driven directly over Wire without an external library.
18 */
19

32 /**

38 /**
39 * @brief Probe the AS5600, verify the magnet, enter low-power mode.
40 *
41 * The AS5600 has no enable pin and stays powered through deep sleep:
42 * in its default mode it draws ~6.5 mA continuously, which alone
43 * would drain the battery in weeks. begin() switches it to LPM3
44 * (~1.5 mA, 10 ms internal sampling — more than enough for wind).
45 * See docs/power-budget.md for why a hardware load switch on the
46 * sensor rail is still the proper long-term fix.
47 *
48 * @return true if the chip answers and the MD status bit is set.
49 */
50 bool begin() override;
51
52 /**
53 * @brief Read the vane angle and publish "wd" [deg, 0-359].
54 * @param out Root JSON object of the telemetry message.
55 * @return true if the angle was read successfully.
56 */
57 bool read(JsonObject &out) override;
58
59private:
60 /**
61 * @brief Read consecutive AS5600 registers.
62 * @param reg First register address.
63 * @param buf Destination buffer.
64 * @param len Number of bytes to read.
65 * @return true if the I2C transaction succeeded.
66 */
67 bool readRegisters(uint8_t reg, uint8_t *buf, size_t len);

73 static constexpr uint8_t kConfPowerModeMask = 0x03; ///< PM bits (CONF low byte).

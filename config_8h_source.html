15// ---------------------------------------------------------------------------
16// LoRa SX1276 868 MHz (SPI)
17// ---------------------------------------------------------------------------
18#define LORA_SCK GPIO_NUM_12 ///< SPI clock
19#define LORA_MISO GPIO_NUM_13 ///< SPI MISO
20#define LORA_MOSI GPIO_NUM_11 ///< SPI MOSI
21#define LORA_CS GPIO_NUM_10 ///< SPI chip select (NSS)
22#define LORA_RST GPIO_NUM_9 ///< Radio reset
23#define LORA_DIO0 GPIO_NUM_8 ///< TX/RX done interrupt
24
25// ---------------------------------------------------------------------------
26// I2C bus (BME280, AS5600, INA3221)
27// The ESP32-S3 Arduino defaults (SDA=8, SCL=9) clash with LORA_DIO0/LORA_RST,
28// so the bus is remapped to GPIO 4/5.
29// ---------------------------------------------------------------------------
30#define I2C_SDA GPIO_NUM_4 ///< I2C data
31#define I2C_SCL GPIO_NUM_5 ///< I2C clock
32
33/// BME280 temperature/humidity/pressure sensor I2C address.
34/// (Named *_I2C_ADDR because the Adafruit driver already defines
35/// BME280_ADDRESS as its 0x77 default.)
36#define BME280_I2C_ADDR 0x76
37/// AS5600 magnetic encoder (wind direction) I2C address.
38#define AS5600_ADDRESS 0x36
39/// INA3221 3-channel current monitor (panel/battery/load) I2C address.
40#define INA3221_ADDRESS 0x40
41
42// ---------------------------------------------------------------------------
43// Pulse sensors (reed switches)
44// ---------------------------------------------------------------------------
45#define RAIN_GAUGE_PIN GPIO_NUM_6 ///< Tipping-bucket rain gauge (RTC-capable, used for wake-on-rain)
46#define ANEMOMETER_PIN GPIO_NUM_7 ///< Cup anemometer reed switch
47
48// ---------------------------------------------------------------------------
49// Misc
50// ---------------------------------------------------------------------------
51#define STATUS_LED_PIN LED_BUILTIN ///< On-board status LED (validated on hardware)
52
53/// CPU frequency during the wake window [MHz]. 80 MHz roughly halves the
54/// active current vs the default 240 MHz; nothing in the cycle (I2C, SPI,
55/// UART, LoRa) needs more.
56#define CPU_FREQ_MHZ 80
57
58#endif // WEATHERSTATION_CONFIG_H

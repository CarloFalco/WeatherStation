1/**
2 * @file main.cpp
3 * @brief WeatherStation V2 firmware entry point.
4 *
5 * Increment 8: reliable delivery. A timer wake-up runs the full cycle:
6 * banner -> configuration -> sensor readings -> telemetry JSON -> LoRa
7 * transmission with ACK window and retries -> radio to sleep -> deep
8 * sleep. The rain accumulator is consumed only on confirmed delivery,
9 * so no rainfall is lost to dropped packets. A rain-pulse wake-up (EXT0)
10 * takes the quick path instead: count the bucket tip in RTC RAM and go
11 * straight back to sleep for the remaining time. loop() is never reached.
12 */

50/**
51 * @brief Listen for the base ACK matching the given sequence number.
52 *
53 * Keeps the receive window open for [lora] ack_timeout_ms, discarding
54 * unrelated packets (other stations, foreign traffic) until the matching
55 * `{"type":"ack","id":<ours>,"seq":<seq>}` arrives or the window closes.
56 *
57 * @param seq Sequence number of the telemetry message just sent.
58 * @param ackOut Filled with the full ACK document on success — the base
59 * may piggyback an OTA offer in it (docs/lora-protocol.md).
60 * @return true if the matching ACK was received.
61 */
62static bool waitForAck(uint16_t seq, JsonDocument &ackOut) {
63 uint32_t deadline = millis() + appConfig.lora.ackTimeoutMs;
64
65 while ((int32_t)(deadline - millis()) > 0) {
66 String rx;
67 if (!telemetryLink.receive(rx, deadline - millis())) {
68 break; // window expired with no packet
69 }
70
71 if (deserializeJson(ackOut, rx) != DeserializationError::Ok) {
72 log_w("RX window: non-JSON packet ignored");
73 continue;
74 }
75 if (strcmp(ackOut["type"] | "", "ack") == 0 &&
76 appConfig.station.id == (ackOut["id"] | "") &&
77 (uint16_t)(ackOut["seq"] | 0) == seq) {
78 return true;
79 }
80 log_d("RX window: unrelated packet ignored (%s)", rx.c_str());
81 }
82 return false;
83}

100void setup() {
101 // First thing: drop the CPU from the default 240 MHz. It roughly halves
102 // the active current and everything in the cycle runs fine at 80 MHz.
103 setCpuFrequencyMhz(CPU_FREQ_MHZ);
104
105 power.begin();
106
107 // --- Quick path: rain pulse during deep sleep ---------------------------
108 // Count the tip and go back to sleep for the time left until the next
109 // scheduled cycle. No serial wait, no sensors, no radio: a rain event
110 // costs milliseconds of CPU, not a full measurement cycle. The RTC
111 // clock keeps running in deep sleep, so time() is monotonic here.
112 if (power.wakeupCause() == ESP_SLEEP_WAKEUP_EXT0) {
113 RainGauge::countSleepPulse();
114 int64_t remaining = (int64_t)g_rtcState.nextWakeEpochS - (int64_t)time(nullptr);
115 if (remaining > 3) {
116 power.deepSleep((uint32_t)remaining);
117 }
118 // Almost time for the scheduled cycle anyway: fall through and run it.
119 }
120
121 Serial.begin(115200);
122#if CORE_DEBUG_LEVEL >= 3
123 // Development builds only: wait for USB-CDC enumeration so the whole
124 // log is visible. Production builds skip it (2 s of awake time saved
125 // per cycle, nobody is listening in the field).
126 delay(2000);
127#endif
128
129 Serial.println();
130 Serial.println("============================================");
131 Serial.printf(" WeatherStation V2 - fw v%s\n", FW_VERSION);
132 Serial.println("============================================");
133 Serial.printf("Boot count : %lu%s\n", (unsigned long)g_rtcState.bootCount,
134 power.isColdBoot() ? " (cold boot)" : "");
135 Serial.printf("Wake-up cause : %s\n", power.wakeupCauseString());
136 Serial.println();
137
138 appConfig.begin();
139 appConfig.printTo(Serial);
140
141 // --- Measurement window -------------------------------------------------
142 Wire.begin(I2C_SDA, I2C_SCL);
143
144 rainGauge.configure(appConfig.rain.mmPerPulse);
145 anemometer.configure(appConfig.wind.mpsPerHz, appConfig.wind.sampleWindowS);
146 windVane.configure(appConfig.wind.vaneOffsetDeg);
147 powerMonitor.configure(appConfig.power.shuntMohm);
148 sensors.add(&bme280);
149 sensors.add(&rainGauge);
150 sensors.add(&anemometer);
151 sensors.add(&windVane);
152 sensors.add(&powerMonitor);
153 size_t healthy = sensors.beginAll();
154 Serial.printf("\nSensors ready : %u\n", (unsigned)healthy);
155
156 JsonDocument doc;
157 JsonObject msg = doc.to<JsonObject>();
158 msg["type"] = "data";
159 msg["id"] = appConfig.station.id;
160 msg["fw"] = FW_VERSION;
161 msg["seq"] = ++g_rtcState.msgSeq;
162 sensors.readAll(msg);
163
164 String payload;
165 serializeJson(doc, payload);
166 Serial.printf("Telemetry (%u bytes): %s\n",
167 (unsigned)payload.length(), payload.c_str());
168
169 // --- Transmission with ACK and retries ----------------------------------
170 uint16_t seq = doc["seq"];
171 bool delivered = false;
172 JsonDocument ackDoc;
173 if (telemetryLink.begin()) {
174 for (uint8_t attempt = 0; attempt <= appConfig.lora.txRetries; attempt++) {
175 if (attempt > 0) {
176 Serial.printf("LoRa TX: retry %u/%u\n", attempt, appConfig.lora.txRetries);
177 }
178 if (!telemetryLink.send(payload)) {
179 continue;
180 }
181 if (!appConfig.lora.ackEnabled) {
182 delivered = true; // fire-and-forget mode: assume delivery
183 break;
184 }
185 if (waitForAck(seq, ackDoc)) {
186 delivered = true;
187 break;
188 }
189 }
190 }
191 Serial.printf("LoRa TX: %s\n", delivered
192 ? (appConfig.lora.ackEnabled ? "delivered (ACK)" : "sent (no-ACK mode)")
193 : "NOT delivered");
194
195 // --- OTA offer piggybacked in the ACK ------------------------------------
196 // Stage 1 (fw 3.0.0-alpha): receive a test image to LittleFS and verify
197 // its CRC-32. Later stages write to the OTA partition and reboot.
198 if (delivered && ackDoc["ota"].is<JsonObject>()) {
199 OtaReceiver::Offer offer;
200 // NOTE: the default must be unsigned. With a plain `| 0` literal
201 // ArduinoJson converts through int, and any CRC above 0x7FFFFFFF
202 // (half of them) would silently read back as 0.
203 offer.size = ackDoc["ota"]["size"] | 0UL;
204 offer.crc = ackDoc["ota"]["crc"] | 0UL;
205 offer.chunks = (uint16_t)(ackDoc["ota"]["chunks"] | 0UL);
206 offer.version = (const char *)(ackDoc["ota"]["ver"] | "");
207 OtaReceiver ota(telemetryLink, appConfig.station.id);
208 ota.configure(appConfig.ota.chunkTimeoutMs, appConfig.ota.maxRetries,
209 appConfig.ota.sessionTimeoutS);
210 ota.runSession(offer);
211 }
212
213 telemetryLink.sleep(); // SX1276 to sleep mode before the MCU powers down
214
215 // Consume the reported rainfall only when delivery is confirmed:
216 // otherwise it stays in the accumulator and rides the next message.
217 if (delivered) {
218 rainGauge.resetAccumulator();
219 }
220
221 blink(3);
222
223 // --- Back to sleep ------------------------------------------------------
224 g_rtcState.nextWakeEpochS = (uint64_t)time(nullptr) + appConfig.station.wakeIntervalS;
225 Serial.printf("\nEntering deep sleep for %lu s\n",
226 (unsigned long)appConfig.station.wakeIntervalS);
227 power.deepSleep(appConfig.station.wakeIntervalS);
228}

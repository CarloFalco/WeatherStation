1/**
2 * @file rtc_state.h
3 * @brief State preserved in RTC slow memory across deep-sleep cycles.
4 *
5 * Variables placed in RTC RAM survive deep sleep but are reset to their
6 * initializers on a power-on reset (cold boot). Everything the station
7 * must remember between wake-ups without touching flash lives here:
8 * flash writes are slow, power-hungry and wear the chip out, RTC RAM
9 * is free.
10 */
11
12#ifndef WEATHERSTATION_RTC_STATE_H
13#define WEATHERSTATION_RTC_STATE_H

15#include <Arduino.h>

17/**
18 * @brief OTA transfer progress, preserved across deep sleep.
19 *
20 * A full firmware image takes ~20 minutes over LoRa: without this, any
21 * interruption would restart the transfer from chunk 0. The station
22 * resumes from @ref nextChunk as soon as the base offers the same image
23 * again (matching size/crc/chunks).
24 */
25struct RtcOtaState {
26 uint32_t size; ///< Expected image size [bytes]; 0 = no transfer pending.
27 uint32_t crc; ///< Expected CRC-32 of the whole image.
28 uint16_t chunks; ///< Total number of chunks of the image.
29 uint16_t nextChunk; ///< First chunk still to receive (resume point).
30 uint32_t runningCrc; ///< CRC-32 accumulator over the chunks already stored.
31};
32
33/**
34 * @brief Persistent station state, kept in RTC RAM between deep sleeps.
35 */
36struct RtcState {
37 uint32_t bootCount; ///< Wake-up cycles since power-on (1 = cold boot).
38 uint16_t msgSeq; ///< LoRa telemetry sequence counter ("seq" JSON field).
39 uint32_t rainPulses; ///< Tipping-bucket pulses accumulated since the last
40 ///< acknowledged transmission.
41 uint64_t nextWakeEpochS; ///< Scheduled time of the next measurement cycle
42 ///< [s, RTC epoch]. The RTC clock keeps running in
43 ///< deep sleep, so after a rain wake-up the station
44 ///< can go back to sleep for the remaining time
45 ///< instead of restarting the full interval.
46 RtcOtaState ota; ///< OTA transfer progress (resume across cycles).
47};
48
49/// Global RTC-resident state instance (defined in PowerManager.cpp).
50extern RtcState g_rtcState;
